class Solution {
  public int maxDotProduct(int[] nums1, int[] nums2) {
    final int m = nums1.length;
    final int n = nums2.length;
    // dp[i][j] := the maximum dot product of the two subsequences nums[0..i)
    // and nums2[0..j)
    int[][] dp = new int[m + 1][n + 1];
    Arrays.stream(dp).forEach(A -> Arrays.fill(A, Integer.MIN_VALUE));

    for (int i = 0; i < m; ++i)
      for (int j = 0; j < n; ++j)
        dp[i + 1][j + 1] = Math.max(Math.max(dp[i][j + 1], dp[i + 1][j]),
                                    Math.max(0, dp[i][j]) + nums1[i] * nums2[j]);

    return dp[m][n];
  }
}

    # dp[i][j] := the maximum dot product of the two subsequences nums[0..i)

/**
 * @param {number[]} A
 * @param {number[]} B
 * @return {number}
 */
var maxDotProduct = function(A, B) {
    const m = A.length;
    const n = B.length;

    // dp[i][j] = maximum dot product of subsequences
    // A[0..i) and B[0..j)
    const dp = Array.from({ length: m + 1 }, () =>
        Array(n + 1).fill(-Infinity)
    );

    for (let i = 0; i < m; i++) {
        for (let j = 0; j < n; j++) {
            dp[i + 1][j + 1] = Math.max(
                dp[i][j + 1],
                dp[i + 1][j],
                Math.max(0, dp[i][j]) + A[i] * B[j]
            );
        }
    }

    return dp[m][n];
};
